const calcularDiasDesdeUltimoRelatorio = (dataUltimoRelatorio) => {
  const hoje = new Date();
  const ultimoRelatorio = new Date(dataUltimoRelatorio);
  const diffTime = Math.abs(hoje - ultimoRelatorio);
  const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24)); 
  return diffDays;
};
